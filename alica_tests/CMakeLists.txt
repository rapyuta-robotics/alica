cmake_minimum_required(VERSION 2.8.3)
project(alica_tests)

## Use c++ 11x std
set(CMAKE_CXX_FLAGS "-std=c++11 -g ${CMAKE_CXX_FLAGS}")
if (${CMAKE_EXTRA_GENERATOR} MATCHES "Eclipse CDT4")
	set(CMAKE_CXX_COMPILER_ARG1 "-std=c++11" CACHE STRING "C++ version for eclipse" FORCE)
	set(CMAKE_ECLIPSE_VERSION "4.5" CACHE STRING "Eclipse version" FORCE)
	add_definitions (-DCMAKE_ECLIPSE_GENERATE_SOURCE_PROJECT=TRUE)
endif (${CMAKE_EXTRA_GENERATOR} MATCHES "Eclipse CDT4")

find_package(catkin REQUIRED COMPONENTS 
    roscpp
    system_config
    fsystem
    autodiff
    constraintsolver
    event_handling
    agent_id
    alica_engine
    alica_ros_proxy
    alica_dummy_proxy
)

include_directories(include)

catkin_package(
  INCLUDE_DIRS include autogenerated/include
  LIBRARIES autogen_code 
  CATKIN_DEPENDS agent_id alica_engine alica_ros_proxy alica_dummy_proxy
)

if (CATKIN_ENABLE_TESTING)

    find_package(rostest REQUIRED)
    file(GLOB_RECURSE test_alica_SOURCES "src/test/*.cpp" src/test_alica.cpp)
    add_rostest_gtest(${PROJECT_NAME}-test test/alica_tests.test 
    ${test_alica_SOURCES})

    if(TARGET ${PROJECT_NAME}-test)
        include_directories(include autogenerated/include ${catkin_INCLUDE_DIRS})
        file(GLOB_RECURSE autogen_SOURCES "autogenerated/src/*.cpp")
        add_library(autogen_code
            ${autogen_SOURCES};
        )
        target_link_libraries(autogen_code ${catkin_LIBRARIES})

        target_link_libraries(${PROJECT_NAME}-test ${catkin_LIBRARIES} autogen_code )
    endif()
endif()
