cmake_minimum_required(VERSION 2.8.3)
project(alica_constraintsolver_tests)

## Use c++ 11x std
set(CMAKE_CXX_FLAGS "-std=c++11 ${CMAKE_CXX_FLAGS}")
if (${CMAKE_EXTRA_GENERATOR} MATCHES "Eclipse CDT4")
	set(CMAKE_CXX_COMPILER_ARG1 "-std=c++11" CACHE STRING "C++ version for eclipse" FORCE)
	set(CMAKE_ECLIPSE_VERSION "4.5" CACHE STRING "Eclipse version" FORCE)
	add_definitions (-DCMAKE_ECLIPSE_GENERATE_SOURCE_PROJECT=TRUE)
endif (${CMAKE_EXTRA_GENERATOR} MATCHES "Eclipse CDT4")

find_package(catkin REQUIRED COMPONENTS 
    roscpp
    system_config
    fsystem
    event_handling
    agent_id
    alica_engine
    alica_dummy_proxy
    constraintsolver
    autodiff
    alica_common_config
    alica_solver_interface
)

include_directories(include)

catkin_package(
  INCLUDE_DIRS include autogenerated/include
  LIBRARIES autogen_code 
  CATKIN_DEPENDS agent_id alica_engine alica_dummy_proxy alica_solver_interface constraintsolver autodiff
)

include_directories(include autogenerated/include ${catkin_INCLUDE_DIRS})
file(GLOB_RECURSE autogen_SOURCES "autogenerated/src/*.cpp")
add_library(autogen_code_constraint_solver_tests ${autogen_SOURCES};)
target_link_libraries(autogen_code_constraint_solver_tests ${catkin_LIBRARIES})

if (CATKIN_ENABLE_TESTING)

    find_package(rostest REQUIRED)
    #file(GLOB_RECURSE test_alica_SOURCES "src/test/*.cpp" src/test_alica.cpp)
    set(test_alica_SOURCES 
        src/test_alica.cpp
        src/test/test_alica_gsolver_plan.cpp
    )

    add_rostest_gtest(${PROJECT_NAME}-test test/alica_tests.test ${test_alica_SOURCES})
    target_link_libraries(${PROJECT_NAME}-test ${catkin_LIBRARIES} autogen_code_constraint_solver_tests )
   
endif()
