cmake_minimum_required(VERSION 3.5.1)
project(alica_ros_turtlesim)
add_compile_options(-std=c++17)

##################
## Dependencies ##
##################

set(PKG_DEPS
    roscpp
    alica_engine
    alica_common_config
    alica_ros_proxy
    constraintsolver
    turtlesim
    geometry_msgs
)

set(BOOST_MIN_VERSION "1.71.0")
find_package(Boost ${BOOST_MIN_VERSION} REQUIRED COMPONENTS filesystem system)

find_package(catkin REQUIRED COMPONENTS ${PKG_DEPS})

catkin_package(
    CATKIN_DEPENDS ${PKG_DEPS}
    INCLUDE_DIRS include alica/Expr/include
)


###########
## Build ##
###########

include_directories(
    include
    alica/Expr/include
    ${catkin_INCLUDE_DIRS}
)

# Add the autogenerated cpp files as a library
file(GLOB_RECURSE autogen_SOURCES "alica/Expr/src/*.cpp")

add_executable(${PROJECT_NAME}
    ${autogen_SOURCES}
    src/turtle.cpp
    src/world_model.cpp
    src/base_node.cpp
    src/base.cpp
)

target_link_libraries(${PROJECT_NAME}
                        Boost::system
                        Boost::filesystem
                        ${CMAKE_DL_LIBS}
                        ${catkin_LIBRARIES}
)

#############
## Install ##
#############

install(TARGETS ${PROJECT_NAME}
        RUNTIME DESTINATION ${CATKIN_PACKAGE_BIN_DESTINATION}
)

install(DIRECTORY include/${PROJECT_NAME}/
        DESTINATION ${CATKIN_PACKAGE_INCLUDE_DESTINATION}
)

install(DIRECTORY launch alica
    DESTINATION ${CATKIN_PACKAGE_SHARE_DESTINATION}
)
